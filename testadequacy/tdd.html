import unittest

def fat(number):
    if number in (0, 1):
        return 1

    return number * fat(number - 1)

class TestFatorial(unittest.TestCase):
    def test_fat(self):
        self.assertEqual(fat(0), 1)
        self.assertEqual(fat(1), 1)
        self.assertEqual(fat(5), 120)
